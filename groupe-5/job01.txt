#Job 01

grep -c "INFO" data.txt     =       affiche le nombre de lignes contenant INFO

grep "WARN" data.txt | sort     =       affiche toutes les lignes contenant WARN, triées alphabétiquement

grep "DEBUG" data.txt | sort -u     =       affiche les lignes uniques contenant DEBUG, sans doublon

grep "POST" data.txt | awk '{print $3}'     =       affiche uniquement la 3e colonne des lignes contenant POST

cat data.txt | tr '[:lower:]' '[:upper:]'     =       transforme toutes les lettres minuscules en MAJUSCULES
